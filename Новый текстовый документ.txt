SELECT name_bludo, image_bludo, name_ing, masse FROM `rec` INNER JOIN  `bludo` ON (`rec`.`id_bludo`=`bludo`.`id_bludo`)
 INNER JOIN  `ing` ON (`rec`.`id_ing_r`=`ing`.`id_ing`)
LIMIT 0, 3;
